name: Deploy GitHub Pages

on:
  push:
    branches:
      - main
  workflow_dispatch:

permissions:
  contents: read
  pages: write
  id-token: write

concurrency:
  group: pages
  cancel-in-progress: true

jobs:
  deploy:
    runs-on: ubuntu-latest
    environment:
      name: github-pages
      url: ${{ steps.deployment.outputs.page_url }}
    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: List repo files
        run: ls -la

      - name: Extract out.zip
        run: |
          rm -rf site
          mkdir -p site
          7z x -y out.zip -osite
          ls -la site
          # Se o zip vier com a pasta "out/" dentro, achata para a raiz do site
          if [ -d "site/out" ]; then
            shopt -s dotglob
            mv site/out/* site/
            rmdir site/out
          fi

          ls -la site
          ls -la site/_next || true

      - name: Configure Pages
        uses: actions/configure-pages@v5

      - name: Upload artifact
        uses: actions/upload-pages-artifact@v3
        with:
          path: site

      - name: Deploy to GitHub Pages
        id: deployment
        uses: actions/deploy-pages@v4
